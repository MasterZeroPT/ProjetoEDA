34bool LerInput(char* nomeficheiro, Antena** listaAntenas, char*** matrizOut, int* linhasOut, int* colunasOut) {
35 FILE* fp = fopen(nomeficheiro, "r");
36 if (fp == NULL) {
37 printf("Erro ao abrir o ficheiro %s", nomeficheiro);
38 return false;
39 }
40 int linhas = 0, capacidadeLinhas = 50;
41 int colunas = 0;
42 char** matriz = malloc(capacidadeLinhas * sizeof(char*)); // aloca um array de apontadores
43 if (!matriz) { //verificar se o malloc da matriz funcionou
44 printf("Erro ao alocar memória para a matriz.\n");
45 fclose(fp);
46 return false;
47 }
48
49 *listaAntenas = NULL; //inicializar a lista ligada
50 char* linha = malloc(256 * sizeof(char)); // buffer para temporariamente armazenar cada linha do ficheiro
51 if (!linha) {
52 printf("Erro ao alocar memória para o buffer da linha.\n");
53 fclose(fp);
54 free(matriz);
55 return false;
56 }
57
58 while (fgets(linha, 256, fp)) { // Lê uma linha do ficheiro apontado por 'fp' e armazena no buffer 'linha'.
59 if (linhas >= capacidadeLinhas) { // verifica se a memória alocada é suficiente e realoca para uma memória maior se não for.
60 capacidadeLinhas *= 2; // duplica o tamanho da capacidade de linhas
61 matriz = realloc(matriz, capacidadeLinhas * sizeof(char*));
62 if (!matriz) {
63 printf("Erro ao realocar memória.\n");
64 free(linha);
65 fclose(fp);
66 return false;
67 }
68 }
69
70 // Conta o número de colunas ignorando os espaços
71 for (int i = 0; linha[i] != '\0'; i++) {
72 if (linha[i] != ' ' && linha[i] != '\n') {
73 colunas++;
74 }
75 }
76
77 // Aloca memória para a linha atual
78 matriz[linhas] = malloc((colunas + 1) * sizeof(char)); // "+1" para garantir espaço para o "\0" na memória
79 if (!matriz[linhas]) {
80 printf("Erro ao alocar memória para a linha %d.\n", linhas);
81 free(linha);
82 fclose(fp);
83 return false;
84 }
85
86 // Preenche a linha da matriz ignorando os espaços
87 int colunaAtual = 0;
88 for (int i = 0; linha[i] != '\0'; i++) {
89 if (linha[i] != ' ' && linha[i] != '\n') {
90 matriz[linhas][colunaAtual++] = linha[i]; // Constroi a matriz de forma organizada sem espaços
91 }
92 }
93 matriz[linhas][colunaAtual] = '\0'; // Adiciona o fim da string
94
95 // Processa os caracteres da linha para criar a lista ligada de antenas
96 for (int coluna = 0; coluna < colunas; coluna++) {
97 if (matriz[linhas][coluna] != '.') {
98 Antena* novaAntena = malloc(sizeof(Antena)); // Verifica se a posição especificada na matriz não está vazia ('.'). Se a posição estiver ocupada, aloca memória para uma nova estrutura Antena.
99 if (!novaAntena) {
100 printf("Erro ao alocar memória para a antena.\n");
101 free(linha);
102 fclose(fp);
103 return false;
104 }
105 novaAntena->frequencia = matriz[linhas][coluna];
106 novaAntena->localizacao.linha = linhas;
107 novaAntena->localizacao.coluna = coluna;
108 novaAntena->proximo = *listaAntenas;
109 *listaAntenas = novaAntena;
110 }
111 }
112
113 linhas++;
114 }
115 free(linha); // Liberta o buffer da linha
116 fclose(fp);
117
118 // Retorna os resultados
119 *matrizOut = matriz;
120 *linhasOut = linhas;
121 *colunasOut = colunas;
122
123 return true;
124}

244void InserirEfeitoNefasto(char** matriz, int linhas, int colunas) {
245 // Iterar sobre todas as células da matriz
246 for (int linha1 = 0; linha1 < linhas; linha1++) {
247 for (int coluna1 = 0; coluna1 < colunas; coluna1++) {
248 char antena1 = matriz[linha1][coluna1];
249 if (antena1 == '.' || antena1 == '#') continue; // Ignorar espaços livres e efeitos já aplicados
250
251 // Verificar alinhamento com outras antenas
252 for (int linha2 = 0; linha2 < linhas; linha2++) {
253 for (int coluna2 = 0; coluna2 < colunas; coluna2++) {
254 char antena2 = matriz[linha2][coluna2];
255 if (antena2 == '.' || antena2 == '#' || (linha1 == linha2 && coluna1 == coluna2)) continue;
256
257 // Verificar se as antenas têm a mesma frequência
258 if (antena1 == antena2) {
259 // Verificar alinhamento horizontal, vertical ou diagonal
260 if (linha1 == linha2 || coluna1 == coluna2 || abs(linha1 - linha2) == abs(coluna1 - coluna2)) {
261 // Calcular incrementos para linha e coluna
262 int incLinha = (linha1 == linha2) ? 0 : (linha2 > linha1 ? 1 : -1);
263 int incColuna = (coluna1 == coluna2) ? 0 : (coluna2 > coluna1 ? 1 : -1);
264
265 // Calcular a distância total
266 int distancia = (linha1 == linha2) ? abs(coluna2 - coluna1) : abs(linha2 - linha1);
267
268 // Verificar se a distância é par (para aplicar o efeito nas extremidades)
269 if (distancia % 2 == 0) {
270 // Calcular as extremidades
271 int linhaEfeito1 = linha1 + incLinha;
272 int colunaEfeito1 = coluna1 + incColuna;
273 int linhaEfeito2 = linha2 - incLinha;
274 int colunaEfeito2 = coluna2 - incColuna;
275
276 // Verificar se as extremidades estão dentro dos limites da matriz e são espaços livres
277 if (linhaEfeito1 >= 0 && linhaEfeito1 < linhas &&
278 colunaEfeito1 >= 0 && colunaEfeito1 < colunas &&
279 matriz[linhaEfeito1][colunaEfeito1] == '.') {
280 matriz[linhaEfeito1][colunaEfeito1] = '#';
281 }
282 if (linhaEfeito2 >= 0 && linhaEfeito2 < linhas &&
283 colunaEfeito2 >= 0 && colunaEfeito2 < colunas &&
284 matriz[linhaEfeito2][colunaEfeito2] == '.') {
285 matriz[linhaEfeito2][colunaEfeito2] = '#';
286 }
287 }
288 }
289 }
290 }
291 }
292 }
293 }
